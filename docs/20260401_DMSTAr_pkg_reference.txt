function (mods, registry = NULL, pparams, Dpy = 365.25, DutyCycle = NULL, derive_PModel = TRUE, default_PModel = 1L, ...) 
{
    if (is.null(registry)) 
        registry <- .P_MODEL_BUILDERS_DEFAULT

        stop("All 'mods' must exist as names in 'registry'.")
    }

    slot_pars <- lapply(mods, function(m) {
        build_P_kinetics(mod_type = m, Dpy = Dpy, DutyCycle = DutyCycle, pparams = pparams, ...)

    K1 <- sapply(slot_pars, function(p) p$K1)

    K3 <- sapply(slot_pars, function(p) p$K3)
    Chalf <- sapply(slot_pars, function(p) p$Chalf)
    Z_1 <- sapply(slot_pars, function(p) p$Z1)
    Z_2 <- sapply(slot_pars, function(p) p$Z2)
    Z_3 <- sapply(slot_pars, function(p) p$Z3)
    K2Coef <- sapply(slot_pars, function(p) safe_num(p$K2Coef, 0))
    SeasonalFactor <- first_non_null_finite(lapply(slot_pars, function(p) p$SeasonalFactor), 0)
    Ytrans <- first_non_null_finite(lapply(slot_pars, function(p) p$Ytrans), NA_real_)
    Ysigma <- first_non_null_finite(lapply(slot_pars, function(p) p$Ysigma), NA_real_)
    CZero <- first_non_null_finite(lapply(slot_pars, function(p) p$Czero), 0)
    PModel <- slot_pars[[1]]$PModel
    list(K1 = K1, K2 = K2, K3 = K3, Chalf = Chalf, Z_1 = Z_1, Z_2 = Z_2, Z_3 = Z_3, K2Coef = K2Coef, Kslots = K, SeasonalFactor = SeasonalFactor,

function (net_table, outlet_count = 5L, case_col = "CaseName") 
{
    req <- c(case_col, "Bypass_to", "Release1_to", "Release2_to", "Outflow_to", "Seepage_to")
    miss <- setdiff(req, names(net_table))
    if (length(miss)) 
        stop("net_table missing columns: ", paste(miss, collapse = ", "))
    parse_dest <- function(x) {
        if (is.null(x) || length(x) == 0 || is.na(x) || trimws(as.character(x)) == "") 
            return(NULL)
        x_chr <- trimws(as.character(x))
        if (grepl("^[0-9]+$", x_chr)) {
            k <- as.integer(x_chr)
            if (k < 1 || k > outlet_count) 
                stop("OUTLET number out of range: ", k)
            return(list(to_type = "OUTLET", to_id = k))
        }
        list(to_type = "CASE", to_id = x_chr)
    }
    stream_cols <- c(bypass = "Bypass_to", release1 = "Release1_to", release2 = "Release2_to", outflow = "Outflow_to", seepage = "Seepage_to")
    out <- list()
    k <- 0L
    for (i in seq_len(nrow(net_table))) {
        from_case <- net_table[[case_col]][i]
        for (s in names(stream_cols)) {
            dest <- parse_dest(net_table[[stream_cols[[s]]]][i])
            if (!is.null(dest)) {
                k <- k + 1L
                out[[k]] <- data.frame(from_case = from_case, stream = s, to_type = dest$to_type, to_id = dest$to_id, frac = 1, 
                  lag_days = 0L, stringsAsFactors = FALSE)
            }
        }
    }
    do.call(rbind, out)
}

FUNCTION: compute_DMSTA_kvals
SIGNATURE: compute_DMSTA_kvals(C1000, Cstar, Ks)
SOURCE FILE: C:/Julian_LaCie/_GitHub/DMSTAr/R/P_kinetics.R:39-80
----------------------------------------------------------------
function (C1000, Cstar, Ks) 
{
    K1 <- K2 <- K3 <- 0
    PModel <- 1
    if (!is.na(Cstar) && Cstar < 0) {
        G <- -Cstar
        if (!is.na(G) && !is.na(Ks) && !is.na(C1000) && G > 0 && Ks > 0 && C1000 > 0) {
            K1 <- Ks/G
            K3 <- Ks * C1000/1000
            K2 <- (1 - G)/G * K3
        }
        PModel <- 2
        return(list(K1 = K1, K2 = K2, K3 = K3, PModel = PModel))
    }
    if (!is.na(C1000) && !is.na(Ks) && !is.na(Cstar) && C1000 > 0 && Ks > 0 && Cstar > 0) {
        f <- max(1, C1000 - Cstar)/1000
        K3 <- Ks * f
        K1 <- K3/Cstar
        K2 <- (K3 * K1)/Ks
    }
    list(K1 = K1, K2 = K2, K3 = K3, PModel = PModel)
}

----------------------------------------------------------------
function (A_cell, ttankS, snap_last = TRUE) 
{
    if (A_cell <= 0) {
        return(list(Ntanks = 1L, A_Tank = 0, F_Tank = 1, Fcum = 1))
    }
    if (ttankS < 1) 
        ttankS <- 1
    Ntanks <- max(1L, floor(ttankS))
    frac <- ttankS - Ntanks
    if (frac > 0) 
        Ntanks <- Ntanks + 1L
    A_Tank <- rep(A_cell/max(1, ttankS), Ntanks)
    if (frac > 0) 
        A_Tank[Ntanks] <- A_cell * frac/ttankS
    F_Tank <- A_Tank/A_cell
    Fcum <- cumsum(F_Tank)
    if (snap_last == TRUE) {
        Fcum[Ntanks] <- 1
    }
    list(Ntanks = Ntanks, A_Tank = A_Tank, F_Tank = F_Tank, Fcum = Fcum)
}

function (V, A_cell, Qi, Rain, Et, Qr0, Qr1, Qr2, Zcontrol_t, Zcontrol_t_minus, Zcontrol_t_plus, params, step_index, nsteps, 
    step_frac, Ddt, Qrelease = 0, QrU1 = 0, QrU2 = 0, RecycleQ = 0, has_depth_constraint = FALSE, zcontrol_mode = c("interp", 
        "fixed"), ...) 
{
    interp_option <- if (is.null(params$interp_option)) 
        2L
    else as.integer(params$interp_option)
    zcontrol_mode <- match.arg(zcontrol_mode)
    Bypass <- 0
    Zmin <- params$Zmin/100
    A <- A_cell
    Vmin <- if (is.null(params$Vmin)) 
        Zmin * A
    else params$Vmin
    Q_a <- params$Q_a
    Q_b <- params$Q_b
    Zweir <- params$Zweir/100
    Q_zmin <- params$Q_zmin/100
    Qomax <- params$Qomax
    Qimax <- params$Qimax
    Width <- params$Width
    Bypass_elev <- params$Bypass_elev/100
    Seepout_Rate <- params$Seepout_Rate
    Seepout_Elev <- params$Seepout_Elev/100
    Seepin_Rate <- params$Seepin_Rate
    Seepin_Elev <- params$Seepin_Elev/100
    ShutdownET <- isTRUE(params$ShutdownET)
    force_Q_out <- isTRUE(params$force_Q_out)
    if (!is.null(params$zcontrol_mode)) {
        zcontrol_mode <- match.arg(as.character(params$zcontrol_mode), c("interp", "fixed"))
    }
    Qr_0 <- if (is.null(Qr0) || !is.finite(Qr0)) 
        0
    else as.numeric(Qr0)
    Z <- if (A > 0) 
        V/A
    else 0
    if (zcontrol_mode == "interp") {
        delta <- ((step_index - 1) + step_frac)/nsteps
        Zcont_raw <- dmsta_interp_zcontrol(Zcontrol_t_minus, Zcontrol_t, Zcontrol_t_plus, delta, interp_option)
    }
    else {
        Zcont_raw <- if (!is.null(Zcontrol_t) && is.finite(Zcontrol_t)) 
            Zcontrol_t
        else 0
    }
    Zcont <- max(Zcont_raw + Q_zmin, Zweir, Zmin)
    Vcontrol <- Zcont * A
    Seepin <- Seepin_Rate * max(0, Seepin_Elev - Z) * A
    Seepout <- Seepout_Rate * max(0, Z - Seepout_Elev) * A
    RainVol <- Rain * A
    Etest <- Et
    EtVol <- Etest * A
    AtmoS <- RainVol - EtVol
    Bypass <- 0
    if (is.finite(Bypass_elev) && Bypass_elev > 0 && Z > Bypass_elev) {
        Bypass <- Qi
    }
    else if (is.finite(Qimax) && Qimax > 0 && Qi > Qimax) {
        Bypass <- Qi - Qimax
    }
    else if (is.finite(Qomax) && Qomax < 0) {
        Bypass <- min(-Qomax, Qi)
    }
    Qnet <- Qi + AtmoS + Seepin + RecycleQ - Seepout - Qrelease - Bypass
    Qo <- 0
    Vtrial <- V + Qnet * Ddt
    if (Vtrial < Vmin) {
        if (Seepout > 0) {
            Seepout <- max(Seepout + (Vtrial - Vmin)/Ddt, 0)
            Qnet <- Qi + AtmoS - Seepout + Seepin + RecycleQ - Qrelease - Bypass
            Vtrial <- V + Qnet * Ddt
        }
        if (Vtrial < Vmin && ShutdownET) {
            Etest <- max(Etest + (Vtrial - Vmin)/Ddt/A, 0)
            AtmoS <- (Rain - Etest) * A
            Qnet <- Qi + AtmoS - Seepout + Seepin + RecycleQ - Qrelease - Bypass
            Vtrial <- V + Qnet * Ddt
        }
    }
    else if (force_Q_out && !has_depth_constraint) {
        Qo <- Qr_0
    }
    else if (!is.null(Q_a) && Q_a < 0) {
        Qo <- Qnet
    }
    else if (A > 0 && Z > Zcont) {
        if (!is.null(Q_a) && Q_a == 0) {
            Qo <- max((V - Vcontrol)/Ddt, 0)
        }
        else {
            Qo <- Q_a * Width * (Z - Zweir)^Q_b
        }
        if (!is.null(Qomax) && Qomax > 0 && Qo > Qomax) {
            Qo <- Qomax
        }
    }
    else {
        Qo <- 0
    }
    if (is.finite(Ddt) && Ddt > 0) {
        Vnext <- V + (Qnet - Qo) * Ddt
    }
    else {
        Vnext <- V
    }
    if (is.finite(Vcontrol) && is.finite(Ddt) && Ddt > 0 && is.finite(Qo) && Qo > 0) {
        if (is.finite(Vnext) && Vnext < Vcontrol) {
            Qo <- max(0, Qnet - (Vcontrol - V)/Ddt)
            Vnext <- V + (Qnet - Qo) * Ddt
        }
    }
    Vnew <- Vnext
    dvdt <- Qnet - Qo
    Qot <- Qo + Qrelease
    list(Vnew = Vnew, dvdt = dvdt, Qo = Qo, Qout = Qot, QrU1 = QrU1, QrU2 = QrU2, Etest = Etest, Seepout = Seepout, Seepin = Seepin, 
        Bypass = Bypass, Z = Z, Zcont = Zcont, Zcont_raw = Zcont_raw, zcontrol_mode = zcontrol_mode)
}

function (state, drivers, ppar, constants) 
{

    Fz <- fC <- c(1, 1, 1)
    for (k in 1:3) {
        if (isTRUE(ppar$Z_1[k] > 0 && zZ < ppar$Z_1[k])) {
            Fz[k] <- zZ/ppar$Z_1[k]
        }
        if (isTRUE(ppar$Chalf[k] > 0)) {
            fC[k] <- ppar$Chalf[k]/(ppar$Chalf[k] + min(C, constants$Cmax))
        }
        else {
            fC[k] <- 1
        }
        if (C < ppar$CZero) {
            fC[k] <- 0
        }
        else if (C > 0) {
            fC[k] <- fC[k] * (C - ppar$CZero)/C
        }
        else {
            fC[k] <- 0
        }
    }
    P_uptake <- P_recycle <- P_sed <- 0
    P_direct <- 0
    if (isTRUE(ppar$PModel == 2L)) {
        k1A <- ppar$K1[1] * Fz[1] * fC[1]
        k2A <- ppar$K2[1]
        k3A <- ppar$K3[1]
        P_uptake <- k1A * C
        P_recycle <- k2A * Y
        k3_used <- k3A
    }
    else {
        if (!is.finite(ppar$Ysigma) || ppar$Ysigma <= 0) {
            Ftrans <- 1
        }
        else {
            Ftrans <- 1/(1 + exp(-(Y - ppar$Ytrans)/ppar$Ysigma))
        }
        k1A <- Ftrans * ppar$K1[1] * Fz[1] * fC[1] + (1 - Ftrans) * ppar$K1[2] * Fz[2] * fC[2]
        k2A <- Ftrans * ppar$K2[1] + (1 - Ftrans) * ppar$K2[2]
        k3A <- Ftrans * ppar$K3[1] + (1 - Ftrans) * ppar$K3[2]
        if (isTRUE(ppar$Z_3[1] > 0 && drivers$Z_plant > ppar$Z_2[1])) {
            if (drivers$Z_plant >= ppar$Z_3[1]) {
                fres <- 1
            }
            else {
                fres <- (drivers$Z_plant - ppar$Z_2[1])/(ppar$Z_3[1] - ppar$Z_2[1])
            }
            k1A <- fres * ppar$K1[3] * Fz[3] * fC[3] + (1 - fres) * k1A
            k2A <- fres * ppar$K2[3] + (1 - fres) * k2A
            k3A <- fres * ppar$K3[3] + (1 - fres) * k3A
        }
        P_uptake <- k1A * C * Y
        P_recycle <- k2A * Y * Y
        k3_used <- k3A
    }
    if (is.finite(ppar$SeasonalFactor) && ppar$SeasonalFactor > 0) {
        xJ <- drivers$julian * 2 * pi/365.25
        sF <- 1.03926 + ppar$SeasonalFactor * (0.02944 * cos(xJ) - 0.14851 * sin(xJ))
        P_uptake <- P_uptake * sF
    }
    P_direct <- ppar$K2Coef[1] * C * Fz[1]
    P_sed <- k3_used * Y
    M_outflow <- (drivers$Qo_tank/A_tank + drivers$Seepout/A_cell) * C
    M_inflow <- drivers$Li_tank/A_tank + (constants$DryDepo + drivers$Rain * constants$C_rain) + (drivers$Seepin * constants$seepin_conc/A_cell)
    dMdt <- (M_inflow + P_recycle - M_outflow - P_uptake - P_direct) * A_tank
    dSdt <- (P_uptake - P_recycle - P_sed) * A_tank
    mratiO <- 0.01
    Mo <- drivers$Mo_fix
    So <- drivers$So_fix
    if (Mo > 0 && (Mo + dMdt * drivers$Ddt) <= mratiO * Mo) {
        dMdt2 <- -Mo * (1 - mratiO)/drivers$Ddt
        fF <- M_inflow + P_recycle - M_outflow - dMdt2/A_tank
        if (fF > 0 && (P_uptake + P_direct) > 0) {
            fF <- fF/(P_uptake + P_direct)
            P_uptake <- P_uptake * fF
            P_direct <- P_direct * fF
        }
        dMdt <- dMdt2
        dSdt <- (P_uptake - P_recycle - P_sed) * A_tank
    }
    if (So > 0 && (So + dSdt * drivers$Ddt) <= mratiO * So) {
        dSdt <- -So * (1 - mratiO)/drivers$Ddt
    }
    list(dMdt = dMdt, dSdt = dSdt, flux = list(P_uptake = P_uptake, P_recycle = P_recycle, P_sed = P_sed, P_direct = P_direct), 
        diag = list(C = C, Y = Y, z = zZ))
}

FUNCTION TYPE: internal
FUNCTION: dmsta_euler_step
SIGNATURE: dmsta_euler_step(V, args_base, Dt = 1)
SOURCE FILE: C:/Julian_LaCie/_GitHub/DMSTAr/R/hydro.R:511-606
----------------------------------------------------------------
function (V, args_base, Dt = 1) 
{
    step_index <- 1L
    nsteps <- 1L
    step_frac <- 0.5
    Ddt <- Dt
    A_cell <- args_base$params$A_cell
    Zmin_cm <- args_base$params$Zmin
    if (!is.finite(Zmin_cm)) 
        Zmin_cm <- 0
    Vmin <- if (is.null(args_base$params$Vmin)) 
        (Zmin_cm/100) * A_cell
    else args_base$params$Vmin
    Vo <- V
    Zrel_cm <- args_base$params$Zrelease
    if (!is.finite(Zrel_cm)) 
        Zrel_cm <- 0
    if (Zrel_cm < Zmin_cm) 
        Zrel_cm <- Zmin_cm
    Zrel_m <- Zrel_cm/100
    Qr0 <- if (is.null(args_base$Qr0) || !is.finite(args_base$Qr0)) 
        0
    else as.numeric(args_base$Qr0)
    Qr1 <- if (is.null(args_base$Qr1) || !is.finite(args_base$Qr1)) 
        0
    else as.numeric(args_base$Qr1)
    Qr2 <- if (is.null(args_base$Qr2) || !is.finite(args_base$Qr2)) 
        0
    else as.numeric(args_base$Qr2)
    if (is.finite(A_cell) && A_cell > 0 && Vo <= Zrel_m * A_cell) {
        QrU1_step <- 0
        QrU2_step <- 0
    }
    else {
        QrU1_step <- Qr1
        QrU2_step <- Qr2
    }
    Qrelease_step <- QrU1_step + QrU2_step
    flux <- do.call(dmsta_deriv_flow, c(list(V = V, step_frac = step_frac, Ddt = Ddt, step_index = step_index, nsteps = nsteps, 
        Qrelease = Qrelease_step, QrU1 = QrU1_step, QrU2 = QrU2_step), args_base))
    V_pred <- V + Dt * flux$dvdt
    qout <- flux$Qout
    if (is.finite(Vmin) && is.finite(Dt) && Dt > 0 && V_pred < Vmin) {
        qout <- max(0, qout - (Vmin - V_pred)/Dt)
        V_new <- Vmin
    }
    else {
        V_new <- V_pred
    }
    Sspec <- Qr0 + QrU1_step + QrU2_step
    if (qout > Sspec && qout > 0) {
        f1 <- QrU1_step/qout
        f2 <- QrU2_step/qout
    }
    else if (Sspec > 0) {
        f1 <- QrU1_step/Sspec
        f2 <- QrU2_step/Sspec
    }
    else {
        f1 <- 0
        f2 <- 0
    }
    f0 <- max(0, 1 - f1 - f2)
    list(V_new = V_new, Qout_ts = qout, SeepOut_ts = flux$Seepout, SeepIn_ts = flux$Seepin, Bypass_ts = flux$Bypass, Etest_ts = flux$Etest, 
        Q_treated_ts = qout * f0, Q_rel1_ts = qout * f1, Q_rel2_ts = qout * f2)
}

function (V, inputs, params, method = c("RK4", "Euler"), Nsteps = 4L) 
{
    method <- match.arg(method)
    Dt <- 1/Nsteps
    V_start <- V
    params2 <- params
    if (is.null(params2$interp_option)) 
        params2$interp_option <- 2L

        inputs$RecycleQ <- 0

        params2$zcontrol_mode <- "interp"
    }
    else {
        params2$zcontrol_mode <- match.arg(as.character(params2$zcontrol_mode), c("interp", "fixed"))
    }

    has_rel <- (is.finite(inputs$Qr1) && inputs$Qr1 != 0) || (is.finite(inputs$Qr2) && inputs$Qr2 != 0)
    if (isTRUE(has_rel)) {
        if (!is.null(Zrelease_used) && is.finite(Zrelease_used) && !is.null(params2$Zmin) && is.finite(params2$Zmin) && Zrelease_used < 
            params2$Zmin) {
            Zrelease_used <- params2$Zmin
        }
    }
    if (is.null(Zrelease_used) || !is.finite(Zrelease_used)) 
        Zrelease_used <- 0
    params2$Zrelease <- Zrelease_used/100
    A_cell <- params2$A_cell
    isa_node <- dmsta_is_node(A_cell, params2$IsaNode)
    Zmin_m <- params2$Zmin/100
    Vmin <- A_cell * Zmin_m
    Qi_eff <- params2$Qin_Frac * inputs$Qi
    RecycleQ <- inputs$RecycleQ
    if (is.null(inputs$Zcontrol) || !is.finite(inputs$Zcontrol)) 
        inputs$Zcontrol <- 0
    if (is.null(inputs$Zcontrol_prev) || !is.finite(inputs$Zcontrol_prev)) {
        inputs$Zcontrol_prev <- inputs$Zcontrol
    }
    if (is.null(inputs$Zcontrol_next) || !is.finite(inputs$Zcontrol_next)) {
        inputs$Zcontrol_next <- inputs$Zcontrol
    }
    has_depth_series <- if (!is.null(inputs$has_depth_constraint)) {
        isTRUE(inputs$has_depth_constraint)
    }
    else {
        any(is.finite(inputs$Zcontrol) & inputs$Zcontrol != 0)
    }
    has_Qr0_series <- if (!is.null(inputs$has_Qr0_series)) {
        isTRUE(inputs$has_Qr0_series)
    }
    else {
        isTRUE(params2$force_Q_out) || (is.finite(inputs$Qr0) && inputs$Qr0 != 0)
    }
    has_Qr1_series <- if (!is.null(inputs$has_Qr1_series)) {
        isTRUE(inputs$has_Qr1_series)
    }
    else {
        any(is.finite(inputs$Qr1) & inputs$Qr1 != 0)
    }
    has_Qr2_series <- if (!is.null(inputs$has_Qr2_series)) {
        isTRUE(inputs$has_Qr2_series)
    }
    else {
        any(is.finite(inputs$Qr2) & inputs$Qr2 != 0)
    }
    if (is.null(params2$force_Q_out)) 
        params2$force_Q_out <- isTRUE(has_Qr0_series)
    Qr0 <- if (isTRUE(has_Qr0_series)) 
        inputs$Qr0
    else 0
    Qr1 <- if (isTRUE(has_Qr1_series)) 
        inputs$Qr1
    else 0
    Qr2 <- if (isTRUE(has_Qr2_series)) 
        inputs$Qr2
    else 0
    has_depth_constraint <- isTRUE(has_depth_series)
    if (isa_node) {
        return(dmsta_node_step(Qi = Qi_eff, RecycleQ = RecycleQ, params = params2, Nsteps = Nsteps, Dt = Dt))
    }
    else {
        Qi <- Qi_eff
        Rain <- inputs$Rain
        Et <- inputs$Et
        Zcontrol_t <- inputs$Zcontrol
        Zcontrol_t_minus <- inputs$Zcontrol_prev
        Zcontrol_t_plus <- inputs$Zcontrol_next
        Qr0 <- if (has_Qr0_series) 
            inputs$Qr0
        else 0
        Qr1 <- if (has_Qr1_series) 
            inputs$Qr1
        else 0
        Qr2 <- if (has_Qr2_series) 
            inputs$Qr2
        else 0
        has_depth_constraint <- has_depth_series
        Zcontrol <- inputs$Zcontrol
        interp_option <- if (is.null(params2$interp_option)) 
            2L
        else as.integer(params2$interp_option)
        if (method == "Euler") {
            Dt <- 1
            step_out <- vector("list", Dt)
            args_base <- list(A_cell = A_cell, Qi = Qi, Rain = Rain, Et = Et, RecycleQ = RecycleQ, Qr0 = Qr0, Qr1 = Qr1, 
                Qr2 = Qr2, Zcontrol_t = Zcontrol, Zcontrol_t_minus = Zcontrol_t_minus, Zcontrol_t_plus = Zcontrol_t_plus, 
                params = params2, has_depth_constraint = has_depth_constraint)
            euler_result <- dmsta_euler_step(V, args_base, Dt = 1)
            Vo <- V
            V <- euler_result$V_new
            step_out[[1]] <- list(Vo = Vo, V = V, Dt = Dt, Qin = Qi, Qout = euler_result$Qout_ts, Q_treated = euler_result$Q_treated_ts, 
                Q_rel1 = euler_result$Q_rel1_ts, Q_rel2 = euler_result$Q_rel2_ts, SeepOut = euler_result$SeepOut_ts, SeepIn = euler_result$SeepIn_ts, 
                Bypass = euler_result$Bypass_ts, Etest = euler_result$Etest_ts, Z = V/A_cell)
        }
        else if (method == "RK4") {
            Dt <- 1/Nsteps
            step_out <- vector("list", Nsteps)
            args_base <- list(A_cell = A_cell, Qi = Qi, Rain = Rain, Et = Et, RecycleQ = RecycleQ, Qr0 = Qr0, Qr1 = Qr1, 
                Qr2 = Qr2, Zcontrol_t = Zcontrol, Zcontrol_t_minus = Zcontrol_t_minus, Zcontrol_t_plus = Zcontrol_t_plus, 
                params = params2, nsteps = Nsteps, has_depth_constraint = has_depth_constraint)
            for (k in seq_len(Nsteps)) {
                Vo <- V
                step_res <- dmsta_rk4_step(V, args_base, step_index = k, Dt = Dt)
                V <- step_res$V_new
                step_out[[k]] <- list(step = k, Vo = Vo, V = V, Dt = Dt, Qi_eff = Qi_eff, Qout = step_res$Qout_ts, Q_treated = step_res$Q_treated_ts, 
                  Q_rel1 = step_res$Q_rel1_ts, Q_rel2 = step_res$Q_rel2_ts, SeepOut = step_res$SeepOut_ts, SeepIn = step_res$SeepIn_ts, 
                  Bypass = step_res$Bypass_ts, Etest = step_res$Etest_ts, Z = V/A_cell)
            }
        }
        Qout_day <- sum(vapply(step_out, function(x) x$Qout * x$Dt, 0))
        Q_treated_day <- sum(vapply(step_out, function(x) x$Q_treated * x$Dt, 0))
        Q_rel1_day <- sum(vapply(step_out, function(x) x$Q_rel1 * x$Dt, 0))
        Q_rel2_day <- sum(vapply(step_out, function(x) x$Q_rel2 * x$Dt, 0))
        SeepOut_day <- sum(vapply(step_out, function(x) x$SeepOut * x$Dt, 0))
        SeepIn_day <- sum(vapply(step_out, function(x) x$SeepIn * x$Dt, 0))
        Bypass_day <- sum(vapply(step_out, function(x) x$Bypass * x$Dt, 0))
        EtVol_day <- sum(vapply(step_out, function(x) x$Etest * A_cell * x$Dt, 0))
        RainVol_day <- Rain * A_cell
        NetAtmo_day <- RainVol_day - EtVol_day
        Z_day <- sum(vapply(step_out, function(x) x$Z * x$Dt, 0))/sum(vapply(step_out, function(x) x$Dt, 0))
        V_day <- sum(vapply(step_out, function(x) ((x$Vo + x$V)/2) * x$Dt, 0))
        RecycleVol_day <- RecycleQ * 1
        WB_in <- Qi_eff + RainVol_day + SeepIn_day + RecycleVol_day
        WB_out <- Qout_day + SeepOut_day + EtVol_day + Bypass_day
        WB_err <- (V - V_start) - (WB_in - WB_out)
        WB_rel <- WB_err/max(1e-12, max(WB_in, WB_out))
        results <- list(V_end = V, V_avg = V_day, Qin = Qi_eff, Qout = Qout_day, Q_treated = Q_treated_day, Q_rel1 = Q_rel1_day, 
            Q_rel2 = Q_rel2_day, SeepOut = SeepOut_day, SeepIn = SeepIn_day, Bypass = Bypass_day, RainVol = RainVol_day, 
            EtVol = EtVol_day, NetAtmo = NetAtmo_day, Z_end = if (A_cell > 0) V/A_cell else NA_real_, Z_avg = Z_day)
        water_budget <- list(WB_in = WB_in, WB_out = WB_out, WB_err = WB_err, WB_rel = WB_rel, RainVol = RainVol_day, EtVol = EtVol_day, 
            NetAtmo = NetAtmo_day, in_components = list(Qi_eff = Qi_eff, RainVol = RainVol_day, SeepIn = SeepIn_day, Recycle = RecycleVol_day), 
            out_components = list(Qout = Qout_day, SeepOut = SeepOut_day, EtVol = EtVol_day, Bypass = Bypass_day), dV = V - 
                V_start)
        mass_budget <- NULL
        meta <- list(V_start = V_start, V_end = V, Nsteps = Nsteps, Dt = Dt, Qi_eff = Qi_eff, params_used = params2, inputs_used = inputs, 
            steps = step_out, method = method)
        out <- list(results = results, budgets = list(water = water_budget, mass = mass_budget), meta = meta)
    }
    return(out)
}

FUNCTION: dmsta_flow_day_steps

----------------------------------------------------------------
function (V, inputs, params, method = c("RK4", "Euler"), Nsteps = 4L) 
{
    method <- match.arg(method)
    day <- dmsta_flow_day(V = V, inputs = inputs, params = params, method = method, Nsteps = Nsteps)
    steps <- NULL
    if (!is.null(day$meta) && !is.null(day$meta$steps)) {
        steps <- day$meta$steps
    }
    else if (!is.null(day$steps)) {
        steps <- day$steps
    }
    else {
        steps <- vector("list", 0L)
    }
    qi_eff <- if (!is.null(day$meta$Qi_eff)) 
        day$meta$Qi_eff
    else if (!is.null(day$meta) && !is.null(day$meta$params_used$Qin_Frac) && !is.null(day$meta$inputs_used$Qi)) {
        day$meta$params_used$Qin_Frac * day$meta$inputs_used$Qi
    }
    else {
        NA_real_
    }
    if (length(steps) > 0) {
        steps <- lapply(seq_along(steps), function(k) {
            s <- steps[[k]]
            if (is.null(s$step)) 
                s$step <- k
            if (is.null(s$Qi_eff)) {
                if (!is.null(s$Qin)) 
                  s$Qi_eff <- s$Qin
                else s$Qi_eff <- qi_eff
            }
            if (is.null(s$Q_treated) && !is.null(s$Q_treated_ts)) 
                s$Q_treated <- s$Q_treated_ts
            if (is.null(s$Q_rel1) && !is.null(s$Q_rel1_ts)) 
                s$Q_rel1 <- s$Q_rel1_ts
            if (is.null(s$Q_rel2) && !is.null(s$Q_rel2_ts)) 
                s$Q_rel2 <- s$Q_rel2_ts
            s
        })
    }
    res <- day$results
    wb <- if (!is.null(day$budgets)) 
        day$budgets$water
    else NULL
    out <- day
    out$steps <- steps
    if (!is.null(res)) {
        out$V_end <- res$V_end
        out$V_avg <- res$V_avg
        out$Z_end <- res$Z_end
        out$Z_avg <- res$Z_avg
        out$Qin <- res$Qin
        out$Qout <- res$Qout
        out$Q_treated <- res$Q_treated
        out$Q_rel1 <- res$Q_rel1
        out$Q_rel2 <- res$Q_rel2
        out$SeepOut <- res$SeepOut
        out$SeepIn <- res$SeepIn
        out$Bypass <- res$Bypass
        out$RainVol <- res$RainVol
        out$EtVol <- res$EtVol
        out$NetAtmo <- res$NetAtmo
    }
    if (!is.null(wb)) {
        out$WB_in <- wb$WB_in
        out$WB_out <- wb$WB_out
        out$WB_err <- wb$WB_err
        out$WB_rel <- wb$WB_rel
        if (is.null(out$RainVol)) 
            out$RainVol <- wb$RainVol
        if (is.null(out$EtVol)) 
            out$EtVol <- wb$EtVol
        if (is.null(out$NetAtmo)) 
            out$NetAtmo <- wb$NetAtmo
    }
    return(out)
}

function (V_init = NULL, series, params, method = c("RK4", "Euler"), Nsteps = 4L) 
{
    method <- match.arg(method)
    n <- nrow(series)
    if (n < 1) 
        stop("series has zero rows.")
    if (!("Date" %in% names(series))) 
        stop("series must include a 'Date' column.")
    if (!all(c("Qi", "Rain", "Et", "Zcontrol") %in% names(series))) {
        stop("series must include columns: Qi, Rain, Et, Zcontrol")
    }
    hx <- dmsta_hydroindex(series, params, check_params_names = TRUE)
    has_Qr0_series <- hx$has_Qr0_series
    has_Qr1_series <- hx$has_Qr1_series
    has_Qr2_series <- hx$has_Qr2_series
    has_depth_constraint_series <- hx$has_depth_constraint_series
    if (is.null(V_init)) {
        Zmin_m <- params$Zmin/100
        if (isTRUE(has_depth_constraint_series)) {
            zc1 <- ifelse(is.finite(series$Zcontrol[1]), series$Zcontrol[1], 0)
            Z0_m <- max(zc1, params$Zmin/100)
        }
        else {
            Z0_m <- max(params$Zinit/100, params$Zmin/100)
        }
        V_init <- params$A_cell * Z0_m
    }
    results_df <- data.frame(Date = as.Date(series$Date), V_end = NA_real_, V_avg = NA_real_, Z_end = NA_real_, Z_avg = NA_real_, 
        Qin = series$Qi, Qout = NA_real_, Q_treated = NA_real_, Q_rel1 = NA_real_, Q_rel2 = NA_real_, SeepOut = NA_real_, 
        SeepIn = NA_real_, Bypass = NA_real_, RainVol = NA_real_, EtVol = NA_real_, NetAtmo = NA_real_, stringsAsFactors = FALSE)
    water_df <- data.frame(Date = as.Date(series$Date), WB_in = NA_real_, WB_out = NA_real_, WB_err = NA_real_, WB_rel = NA_real_, 
        stringsAsFactors = FALSE)
    V <- V_init
    params_run <- params
    if (is.null(params_run$force_Q_out)) {
        params_run$force_Q_out <- isTRUE(has_Qr0_series)
    }
    for (i in seq_len(n)) {
        in_i <- as.list(series[i, ])
        if (!has_Qr0_series) 
            in_i$Qr0 <- 0
        nz <- dmsta_zneighbors(i, series$Zcontrol)
        in_i$Zcontrol <- nz$today
        in_i$Zcontrol_prev <- nz$prev_day
        in_i$Zcontrol_next <- nz$nxt
        in_i$has_depth_constraint <- isTRUE(has_depth_constraint_series)
        if (!has_Qr0_series) 
            in_i$Qr0 <- 0
        if (!has_Qr1_series) 
            in_i$Qr1 <- 0
        if (!has_Qr2_series) 
            in_i$Qr2 <- 0
        if (is.null(in_i$RecycleQ)) 
            in_i$RecycleQ <- 0
        day_res <- dmsta_flow_day(V = V, inputs = in_i, params = params_run, Nsteps = Nsteps, method = method)
        V <- day_res$results$V_end
        results_df$Qin[i] <- day_res$results$Qin
        results_df$Qout[i] <- day_res$results$Qout
        results_df$Q_treated[i] <- day_res$results$Q_treated
        results_df$Q_rel1[i] <- day_res$results$Q_rel1
        results_df$Q_rel2[i] <- day_res$results$Q_rel2
        results_df$SeepOut[i] <- day_res$results$SeepOut
        results_df$SeepIn[i] <- day_res$results$SeepIn
        results_df$Bypass[i] <- day_res$results$Bypass
        results_df$V_end[i] <- day_res$results$V_end
        results_df$Z_end[i] <- day_res$results$Z_end
        results_df$V_avg[i] <- day_res$results$V_avg
        results_df$Z_avg[i] <- day_res$results$Z_avg
        results_df$RainVol[i] <- day_res$budgets$water$RainVol
        results_df$EtVol[i] <- day_res$budgets$water$EtVol
        results_df$NetAtmo[i] <- day_res$budgets$water$NetAtmo
        water_df$WB_in[i] <- day_res$budgets$water$WB_in
        water_df$WB_out[i] <- day_res$budgets$water$WB_out
        water_df$WB_err[i] <- day_res$budgets$water$WB_err
        water_df$WB_rel[i] <- day_res$budgets$water$WB_rel
    }
    out <- list(results = results_df, budgets = list(water = water_df, mass = NULL), meta = list(V_init = V_init, V_end = V, 
        Nsteps = Nsteps, method = method, A_cell = params$A_cell, has_depth_constraint_series = has_depth_constraint_series, 
        has_Qr0_series = has_Qr0_series, has_Qr1_series = has_Qr1_series, has_Qr2_series = has_Qr2_series))
    class(out) <- c("dmsta_hydro_result", "list")
    out
}

function (series, params = list(), zero_tol = 0, check_params_names = TRUE, name_fields = list(Qr0 = c("Qr0_name", "Qlabel_0", 
    "Q0_name"), Qr1 = c("Qr1_name", "Qlabel_1", "Q1_name"), Qr2 = c("Qr2_name", "Qlabel_2", "Q2_name"), Z = c("Zcon_name", 
    "Zcontrol_name", "Z_Label", "Z_name")), none_tokens = c("", "<none>", "none", NA_character_, NULL, NA), semantics = c("dmsta", 
    "presence")) 
{
    semantics <- match.arg(semantics)
    any_nz <- function(x) {
        if (is.null(x)) 
            return(FALSE)
        x <- suppressWarnings(as.numeric(x))
        any(is.finite(x) & abs(x) > zero_tol)
    }
    get_label <- function(key) {
        flds <- name_fields[[key]]
        if (is.null(flds) || length(flds) == 0) 
            return(NULL)
        for (nm in flds) {
            if (!is.null(params[[nm]])) {
                v <- params[[nm]]
                if (!is.character(v)) 
                  v <- as.character(v)
                v <- trimws(v)
                v <- v[nzchar(v) & !(tolower(v) %in% tolower(none_tokens))]
                if (length(v) > 0) 
                  return(v[[1]])
            }
        }
        NULL
    }
    classify_label <- function(lbl) {
        if (is.null(lbl) || !is.character(lbl) || !nzchar(lbl)) {
            return(list(any = FALSE, forced = FALSE, observed = FALSE, label = NULL))
        }
        lbl <- trimws(lbl)
        obs <- substr(lbl, 1, 1) == "*"
        list(any = TRUE, forced = !obs, observed = obs, label = lbl)
    }
    s_has_Qr0 <- any_nz(series$Qr0)
    s_has_Qr1 <- any_nz(series$Qr1)
    s_has_Qr2 <- any_nz(series$Qr2)
    s_has_Z <- any_nz(series$Zcontrol)
    lab_Qr0 <- classify_label(get_label("Qr0"))
    lab_Qr1 <- classify_label(get_label("Qr1"))
    lab_Qr2 <- classify_label(get_label("Qr2"))
    lab_Z <- classify_label(get_label("Z"))
    if (semantics == "dmsta") {
        has_Qr0 <- isTRUE(lab_Qr0$forced) || isTRUE(params$force_Q_out)
        has_Qr1 <- isTRUE(lab_Qr1$forced)
        has_Qr2 <- isTRUE(lab_Qr2$forced)
        has_Z <- isTRUE(lab_Z$forced)
    }
    else {
        has_Qr0 <- s_has_Qr0
        has_Qr1 <- s_has_Qr1
        has_Qr2 <- s_has_Qr2
        has_Z <- s_has_Z
        if (isTRUE(check_params_names)) {
            has_Qr0 <- has_Qr0 || lab_Qr0$any || isTRUE(params$force_Q_out)
            has_Qr1 <- has_Qr1 || lab_Qr1$any
            has_Qr2 <- has_Qr2 || lab_Qr2$any
            has_Z <- has_Z || lab_Z$any
        }
    }
    list(has_Qr0_series = has_Qr0, has_Qr1_series = has_Qr1, has_Qr2_series = has_Qr2, has_depth_constraint_series = has_Z, 
        labels = list(Qr0 = lab_Qr0, Qr1 = lab_Qr1, Qr2 = lab_Qr2, Z = lab_Z))

function (Zcontrol1, Zcontrol, Zcontrol2, delta, interp_option = 2L) 
{
    if (is.null(Zcontrol) || !is.finite(Zcontrol)) 
        Zcontrol <- 0
    if (is.null(Zcontrol1) || !is.finite(Zcontrol1)) 
        Zcontrol1 <- Zcontrol
    if (is.null(Zcontrol2) || !is.finite(Zcontrol2)) 
        Zcontrol2 <- Zcontrol
    if (!is.finite(delta)) 
        delta <- 0
    delta <- max(0, min(1, delta))
    switch(as.character(interp_option), `1` = Zcontrol, `2` = {
        if (delta <= 0.5) {
            Zcontrol1 + (Zcontrol - Zcontrol1) * (delta + 0.5)
        } else {
            Zcontrol + (Zcontrol2 - Zcontrol) * (delta - 0.5)
        }
    }, `3` = {
        Zcontrol1 * (1 - delta) + Zcontrol * delta
    }, {
        if (delta <= 0.5) {
            Zcontrol1 + (Zcontrol - Zcontrol1) * (delta + 0.5)
        } else {
            Zcontrol + (Zcontrol2 - Zcontrol) * (delta - 0.5)
        }
    })
}

----------------------------------------------------------------
function (Qi, RecycleQ = 0, Seepout_Rate = 0, Qimax = 0, Qomax = 0) 
{
    qout <- max(0, Qi + RecycleQ)
    Etest <- 0
    seepout <- 0
    seepin <- 0
    bypass <- 0
    if (is.finite(Seepout_Rate) && Seepout_Rate > 0) {
        seepout <- qout
        qout <- 0
    }
    else if (is.finite(Qimax) && Qimax > 0) {
        bypass <- qout
        qout <- 0
    }
    else if (is.finite(Qomax) && Qomax < 0) {
        bypass <- min(Qi, -Qomax)
        bypass <- max(0, bypass)
        qout <- max(0, qout - bypass)
    }
    else if (is.finite(Qomax) && Qomax > 0) {
        bypass <- max(0, qout - Qomax)
        qout <- max(0, qout - bypass)
    }
    list(Qout = qout, Etest = Etest, SeepOut = seepout, SeepIn = seepin, Bypass = bypass)
}

FUNCTION TYPE: internal
FUNCTION: dmsta_rk4_P_step
SIGNATURE: dmsta_rk4_P_step(M, S, args_base, Dt, ppar, constants, clamp = TRUE, Mmin = 0, Smin = 0)
SOURCE FILE: C:/Julian_LaCie/_GitHub/DMSTAr/R/P_engine.R:251-337
----------------------------------------------------------------
function (M, S, args_base, Dt, ppar, constants, clamp = TRUE, Mmin = 0, Smin = 0) 
{
    if (is.null(args_base$drivers) || !is.list(args_base$drivers)) {
        stop("args_base$drivers must be a list.")
    }
    stage_call <- function(Ms, Ss, step_frac, Ddt_use) {
        drv <- args_base$drivers
        drv$StepFrac <- step_frac
        drv$Ddt <- Ddt_use
        drv$Mo_fix <- Mo
        drv$So_fix <- So
        dmsta_deriv_mass(state = list(M = Ms, S = Ss), drivers = drv, ppar = ppar, constants = constants)
    }
    Mo <- M
    So <- S
    s1 <- stage_call(Mo, So, step_frac = 0, Ddt_use = 0.5 * Dt)
    M2 <- Mo + s1$dMdt * 0.5 * Dt
    S2 <- So + s1$dSdt * 0.5 * Dt
    s2 <- stage_call(M2, S2, step_frac = 0.5, Ddt_use = 0.5 * Dt)
    M3 <- Mo + s2$dMdt * 0.5 * Dt
    S3 <- So + s2$dSdt * 0.5 * Dt
    s3 <- stage_call(M3, S3, step_frac = 0.5, Ddt_use = Dt)
    M4 <- Mo + s3$dMdt * Dt
    S4 <- So + s3$dSdt * Dt
    s4 <- stage_call(M4, S4, step_frac = 1, Ddt_use = Dt)
    dMdt_ts <- (s1$dMdt + 2 * s2$dMdt + 2 * s3$dMdt + s4$dMdt)/6
    dSdt_ts <- (s1$dSdt + 2 * s2$dSdt + 2 * s3$dSdt + s4$dSdt)/6
    M_new <- Mo + dMdt_ts * Dt
    S_new <- So + dSdt_ts * Dt
    if (clamp) {
        if (!is.finite(M_new)) 
            M_new <- Mmin
        if (!is.finite(S_new)) 
            S_new <- Smin
        if (M_new < Mmin) 
            M_new <- Mmin
        if (S_new < Smin) 
            S_new <- Smin
    }
    P_uptake_ts <- (s1$flux$P_uptake + 2 * s2$flux$P_uptake + 2 * s3$flux$P_uptake + s4$flux$P_uptake)/6
    P_recycle_ts <- (s1$flux$P_recycle + 2 * s2$flux$P_recycle + 2 * s3$flux$P_recycle + s4$flux$P_recycle)/6
    P_sed_ts <- (s1$flux$P_sed + 2 * s2$flux$P_sed + 2 * s3$flux$P_sed + s4$flux$P_sed)/6
    P_direct_ts <- (s1$flux$P_direct + 2 * s2$flux$P_direct + 2 * s3$flux$P_direct + s4$flux$P_direct)/6
    C_ts <- (s1$diag$C + 2 * s2$diag$C + 2 * s3$diag$C + s4$diag$C)/6
    Y_ts <- (s1$diag$Y + 2 * s2$diag$Y + 2 * s3$diag$Y + s4$diag$Y)/6
    z_ts <- (s1$diag$z + 2 * s2$diag$z + 2 * s3$diag$z + s4$diag$z)/6

        P_sed_ts = P_sed_ts, P_direct_ts = P_direct_ts, C_ts = C_ts, Y_ts = Y_ts, z_ts = z_ts)

FUNCTION: dmsta_rk4_step

function (V, args_base, step_index, Dt) 
{
    A_cell <- args_base$params$A_cell
    Zmin_cm <- args_base$params$Zmin
    if (!is.finite(Zmin_cm)) 
        Zmin_cm <- 0
    Vmin <- if (is.null(args_base$params$Vmin)) 
        (Zmin_cm/100) * A_cell
    else args_base$params$Vmin
    Vo <- V
    Zrel_cm <- args_base$params$Zrelease
    if (!is.finite(Zrel_cm)) 
        Zrel_cm <- 0
    if (Zrel_cm < Zmin_cm) 
        Zrel_cm <- Zmin_cm
    Zrel_m <- Zrel_cm/100
    Qr0 <- if (is.null(args_base$Qr0) || !is.finite(args_base$Qr0)) 
        0
    else as.numeric(args_base$Qr0)
    Qr1 <- if (is.null(args_base$Qr1) || !is.finite(args_base$Qr1)) 
        0
    else as.numeric(args_base$Qr1)
    Qr2 <- if (is.null(args_base$Qr2) || !is.finite(args_base$Qr2)) 
        0
    else as.numeric(args_base$Qr2)
    if (is.finite(A_cell) && A_cell > 0 && Vo <= Zrel_m * A_cell) {
        QrU1_step <- 0
        QrU2_step <- 0
    }
    else {
        QrU1_step <- Qr1
        QrU2_step <- Qr2
    }
    Qrelease_step <- QrU1_step + QrU2_step
    QrU0 <- Qr0
    s1 <- do.call(dmsta_deriv_flow, c(list(V = V, step_frac = 0, Ddt = 0.5 * Dt, step_index = step_index, Qrelease = Qrelease_step, 
        QrU1 = QrU1_step, QrU2 = QrU2_step), args_base))
    V2 <- V + s1$dvdt * Dt/2
    s2 <- do.call(dmsta_deriv_flow, c(list(V = V2, step_frac = 0.5, Ddt = 0.5 * Dt, step_index = step_index, Qrelease = Qrelease_step, 
        QrU1 = QrU1_step, QrU2 = QrU2_step), args_base))
    V3 <- V + s2$dvdt * Dt/2
    s3 <- do.call(dmsta_deriv_flow, c(list(V = V3, step_frac = 0.5, Ddt = Dt, step_index = step_index, Qrelease = Qrelease_step, 
        QrU1 = QrU1_step, QrU2 = QrU2_step), args_base))
    V4 <- V + s3$dvdt * Dt
    s4 <- do.call(dmsta_deriv_flow, c(list(V = V4, step_frac = 1, Ddt = Dt, step_index = step_index, Qrelease = Qrelease_step, 
        QrU1 = QrU1_step, QrU2 = QrU2_step), args_base))
    dv <- (s1$dvdt + 2 * s2$dvdt + 2 * s3$dvdt + s4$dvdt)/6
    qout <- (s1$Qout + 2 * s2$Qout + 2 * s3$Qout + s4$Qout)/6
    seepO <- (s1$Seepout + 2 * s2$Seepout + 2 * s3$Seepout + s4$Seepout)/6
    seepI <- (s1$Seepin + 2 * s2$Seepin + 2 * s3$Seepin + s4$Seepin)/6
    byps <- (s1$Bypass + 2 * s2$Bypass + 2 * s3$Bypass + s4$Bypass)/6
    Etest_m <- (s1$Etest + 2 * s2$Etest + 2 * s3$Etest + s4$Etest)/6
    V_pred <- V + dv * Dt
    if (is.finite(Vmin) && is.finite(Dt) && Dt > 0 && V_pred < Vmin) {
        qout <- max(0, qout - (Vmin - V_pred)/Dt)
        V_new <- Vmin
    }
    else {
        V_new <- V_pred
    }
    q_end <- qout
    Sspec <- QrU0 + QrU1_step + QrU2_step
    if (q_end > Sspec && q_end > 0) {
        f1 <- QrU1_step/q_end
        f2 <- QrU2_step/q_end
    }
    else if (Sspec > 0) {
        f1 <- QrU1_step/Sspec
        f2 <- QrU2_step/Sspec
    }
    else {
        f1 <- 0
        f2 <- 0
    }
    f0 <- max(0, 1 - f1 - f2)
    list(V_new = V_new, Qout_ts = qout, SeepOut_ts = seepO, SeepIn_ts = seepI, Bypass_ts = byps, Etest_ts = Etest_m, Q_treated_ts = qout * 
        f0, Q_rel1_ts = qout * f1, Q_rel2_ts = qout * f2)
}

function (cells) 
{
    if (!is.list(cells) || length(cells) < 1) 
        stop("'cells' must be a non-empty list.")
    n <- length(cells)
    for (i in seq_len(n)) {
        cd <- cells[[i]]
        if (is.null(cd$params)) 
            stop("cells[[", i, "]]$params is missing.")
        if (is.null(cd$ttankS)) 
            stop("cells[[", i, "]]$ttankS is missing.")
        if (is.null(cd$ppar)) 
            stop("cells[[", i, "]]$ppar is missing. Build per-cell kinetics with build_P_kin_slots().")
        if (is.null(cd$constants)) 
            stop("cells[[", i, "]]$constants is missing. Build per-cell constants list (Cmax, C_rain, DryDepo, etc.).")
        need_ppar <- c("K1", "K2", "K3", "Z_1", "Z_2", "Z_3", "Chalf", "K2Coef", "PModel", "SeasonalFactor", "CZero")
        miss_ppar <- setdiff(need_ppar, names(cd$ppar))
        if (length(miss_ppar) > 0) {
            stop("cells[[", i, "]]$ppar missing fields: ", paste(miss_ppar, collapse = ", "))
        }
        need_const <- c("Cmax", "C_rain", "DryDepo", "seepin_conc", "seepout_conc_max", "fseep_recycle", "fseep_out")
        miss_const <- setdiff(need_const, names(cd$constants))
        if (length(miss_const) > 0) {
            stop("cells[[", i, "]]$constants missing fields: ", paste(miss_const, collapse = ", "))
        }
        if (is.null(cd$DownCell)) 
            cd$DownCell <- 0L
        if (is.null(cd$Qin_Frac)) 
            cd$Qin_Frac <- 0
        if (is.null(cd$RecycleIndex) || is.na(cd$RecycleIndex) || cd$RecycleIndex == 0L) {
            cd$RecycleIndex <- i
        }
        cells[[i]] <- cd
    }
    spl <- which(vapply(cells, function(x) identical(toupper(x$label), "SPLITTER"), logical(1)))
    if (length(spl) > 1) 
        stop("Only one SPLITTER cell is supported (matches DMSTA).")
    if (length(spl) == 1 && is.null(cells[[spl]]$SplitterFrac)) {
        stop("Splitter cell must provide SplitterFrac.")
    }
    cells
}

                }
                if (is.data.frame(res$results$cell[[cell_index]])) {
                  return(res$results$cell[[cell_index]])

function (cases, net_table = NULL, routes = NULL, outlet_count = 5L, verbose = TRUE, check_route = FALSE, Nsteps_case = NULL,

{

    }

        routes <- build_routes_from_net_table(net_table, outlet_count = outlet_count)
    }
    bad_from <- setdiff(unique(routes$from_case), case_names)
    if (length(bad_from)) 
        stop("Unknown from_case in routes: ", paste(bad_from, collapse = ", "))
    bad_to <- setdiff(unique(routes$to_id[routes$to_type == "CASE"]), case_names)
    if (length(bad_to))

    outlets <- lapply(seq_len(outlet_count), function(i) list(Q = numeric(nday), L = numeric(nday)))
    names(outlets) <- paste0("Outlet", seq_len(outlet_count))
    case_results <- vector("list", length(case_names))
    names(case_results) <- case_names
    ledger <- data.frame(from_case = character(0), stream = character(0), to_type = character(0), to_id = character(0), frac = numeric(0), 
        lag_days = integer(0), total_Q = numeric(0), total_L = numeric(0), stringsAsFactors = FALSE)
    dots <- list(...)
    nsteps_default <- if (!is.null(dots$Nsteps)) 
        as.integer(dots$Nsteps)
    else 4L
    dots$Nsteps <- NULL
    qmethod_default <- if (!is.null(dots$Qmethod)) 
        as.character(dots$Qmethod)
    else "RK4"
    dots$Qmethod <- NULL
    resolve_nsteps <- function(case_id) {
        if (is.null(Nsteps_case)) 
            return(nsteps_default)
        if (length(Nsteps_case) == 1L && is.null(names(Nsteps_case))) 
            return(as.integer(Nsteps_case))
        if (!is.null(names(Nsteps_case)) && case_id %in% names(Nsteps_case)) {
            return(as.integer(Nsteps_case[[case_id]]))
        }
        nsteps_default
    }
    resolve_qmethod <- function(case_id) {
        if (is.null(Qmethod_case)) 
            return(qmethod_default)
        if (length(Qmethod_case) == 1L && is.null(names(Qmethod_case))) 
            return(as.character(Qmethod_case))
        if (!is.null(names(Qmethod_case)) && case_id %in% names(Qmethod_case)) {
            return(as.character(Qmethod_case[[case_id]]))
        }
        qmethod_default
    }
    for (cn in exec_order) {
        cd <- cases[[cn]]
        base <- cd$series_base
        needed <- c("Date", "Qi", "Ci", "Rain", "Et", "Zcontrol")
        if (!all(needed %in% names(base))) {
            stop("Case ", cn, " series_base must include: ", paste(needed, collapse = ", "))
        }
        if (length(base$Date) != nday) 
            stop("Case ", cn, " Date length differs; expected shared network calendar.")
        Qb <- base$Qi
        Cb <- base$Ci
        Qb[!is.finite(Qb)] <- 0
        Cb[!is.finite(Cb)] <- 0
        Lb <- Qb * Cb
        rq <- routed_in[[cn]]$Q
        rl <- routed_in[[cn]]$L
        rq[!is.finite(rq)] <- 0
        rl[!is.finite(rl)] <- 0
        Qin <- Qb + rq
        Lin <- Lb + rl
        Cin <- ifelse(Qin > 0, Lin/Qin, 0)
        Cin[!is.finite(Cin)] <- 0
        series_run <- base
        series_run$Qi <- Qin
        series_run$Ci <- Cin
        nsteps_i <- resolve_nsteps(cn)
        qmethod_i <- resolve_qmethod(cn)
        if (verbose) 
            message("Running CASE: ", cn, " (Nsteps=", nsteps_i, ", Qmethod=", qmethod_i, ")")
        if (verbose) {
            message(sprintf("CASE %s: baseQ sum=%.4f anyNA=%s | routedQ sum=%.4f anyNA=%s | routedL sum=%.4f anyNA=%s", cn, 
                sum(base$Qi, na.rm = TRUE), anyNA(base$Qi), sum(routed_in[[cn]]$Q, na.rm = TRUE), anyNA(routed_in[[cn]]$Q), 
                sum(routed_in[[cn]]$L, na.rm = TRUE), anyNA(routed_in[[cn]]$L)))
        }
        if (cn == "FEB_S5A") {
            assign(".debug_series_run", series_run, envir = .GlobalEnv)
            assign(".debug_cells", cd$cells, envir = .GlobalEnv)
            assign(".debug_routed_in", routed_in[[cn]], envir = .GlobalEnv)
        }
        res <- do.call(dmsta_flowP_case, c(list(series = series_run, cells = cd$cells, Nsteps = nsteps_i, Qmethod = qmethod_i), 
            dots))
        case_results[[cn]] <- res
        df <- extract_df(res, level = "case")
        if (check_route) {
            message(cn, " sums:", " bypass=", sum(df$Q_out_bypass, na.rm = TRUE), " treated=", sum(df$Q_out_treated, na.rm = TRUE), 
                " rel1=", sum(df$Q_out_release1, na.rm = TRUE), " rel2=", sum(df$Q_out_release2, na.rm = TRUE), " seep=", 
                sum(df$Q_out_seep_discharge, na.rm = TRUE), " total=", sum(df$Q_out_total, na.rm = TRUE))
        }
        rsub <- routes[routes$from_case == cn, , drop = FALSE]
        if (nrow(rsub)) {
            for (ri in seq_len(nrow(rsub))) {
                r <- rsub[ri, ]
                cols <- stream_map_cols(r$stream, outflow_def = "treated")
                if (!(cols["Q"] %in% names(df) && cols["L"] %in% names(df))) {
                  stop("Case ", cn, " output missing: ", cols["Q"], " / ", cols["L"])
                }
                Qs <- df[[cols["Q"]]] * r$frac
                Ls <- df[[cols["L"]]] * r$frac
                Qs[!is.finite(Qs)] <- 0
                Ls[!is.finite(Ls)] <- 0
                if (!is.null(r$lag_days) && r$lag_days != 0L) {
                  Qs <- lag_vec(Qs, r$lag_days)
                  Ls <- lag_vec(Ls, r$lag_days)
                }
                if (r$to_type == "CASE") {
                  to_case <- as.character(r$to_id)
                  routed_in[[to_case]]$Q <- routed_in[[to_case]]$Q + Qs
                  routed_in[[to_case]]$L <- routed_in[[to_case]]$L + Ls
                }
                else {
                  k <- as.integer(r$to_id)
                  outlets[[k]]$Q <- outlets[[k]]$Q + Qs
                  outlets[[k]]$L <- outlets[[k]]$L + Ls
                }
                ledger <- rbind(ledger, data.frame(from_case = cn, stream = r$stream, to_type = r$to_type, to_id = as.character(r$to_id), 
                  frac = r$frac, lag_days = as.integer(r$lag_days), total_Q = sum(Qs, na.rm = TRUE), total_L = sum(Ls, na.rm = TRUE), 
                  stringsAsFactors = FALSE))
            }
        }
    }
    outlet_summary <- do.call(rbind, lapply(seq_len(outlet_count), function(k) {
        Q <- outlets[[k]]$Q
        L <- outlets[[k]]$L
        data.frame(outlet = paste0("Outlet ", k), total_Q = sum(Q, na.rm = TRUE), total_L = sum(L, na.rm = TRUE), FWC = fw(sum(L, 
            na.rm = TRUE), sum(Q, na.rm = TRUE)), stringsAsFactors = FALSE)
    }))
    list(order = exec_order, case_results = case_results, routed_in = routed_in, outlets = outlets, outlet_summary = outlet_summary, 
        ledger = ledger)
}

function (stream, outflow_def = c("treated", "total")) 
{
    outflow_def <- match.arg(outflow_def)
    stream <- tolower(stream)
    if (stream == "outflow") {
        if (outflow_def == "treated") {
            return(c(Q = "Q_out_treated", L = "L_out_treated"))
        }
        else {
            return(c(Q = "Q_out_total", L = "L_out_total"))
        }
    }
    switch(stream, bypass = c(Q = "Q_out_bypass", L = "L_out_bypass"), release1 = c(Q = "Q_out_release1", L = "L_out_release1"), 
        release2 = c(Q = "Q_out_release2", L = "L_out_release2"), seepage = c(Q = "Q_out_seep_discharge", L = "L_out_seep_discharge"), 
        stop("Unknown stream: ", stream))
}

FUNCTION TYPE: internal
FUNCTION: topo_order_cases
SIGNATURE: topo_order_cases(routes, case_names)
SOURCE FILE: C:/Julian_LaCie/_GitHub/DMSTAr/R/Network_cases.R:299-329
----------------------------------------------------------------
function (routes, case_names) 
{
    edges <- routes[routes$to_type == "CASE", c("from_case", "to_id"), drop = FALSE]
    names(edges) <- c("from", "to")
    edges <- unique(edges)
    incoming <- setNames(rep(0L, length(case_names)), case_names)
    adj <- setNames(vector("list", length(case_names)), case_names)
    for (i in seq_len(nrow(edges))) {
        a <- edges$from[i]
        b <- edges$to[i]
        adj[[a]] <- unique(c(adj[[a]], b))
        incoming[[b]] <- incoming[[b]] + 1L
    }
    q <- names(incoming[incoming == 0L])
    out <- character(0)
    while (length(q)) {
        n <- q[1]
        q <- q[-1]
        out <- c(out, n)
        for (m in adj[[n]]) {
            incoming[[m]] <- incoming[[m]] - 1L
            if (incoming[[m]] == 0L) 
                q <- c(q, m)
        }
    }
    if (length(out) != length(case_names)) {
        stop("Network contains a cycle; strictly downstream requirement violated.")
    }
    out
}
